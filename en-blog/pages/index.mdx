## Stats

export const Stats = () => {
  const fetcher = (...args) => fetch(...args).then((res) => res.json())
  const { data: github } = useSWR('/api/github', fetcher);
  const { data: youtube } = useSWR('/api/youtube', fetcher);
  return (<ul>
    <li>GitHub Followers - <a href="https://github.com/geekplux">{github?.followers || 'fetching...'}</a></li>
    <li>GitHub Stars - <a href="https://github.com/geekplux">{github?.stars || 'fetching...'}</a></li>
    <li>YouTube Subscribers - <a href="https://youtube.com/c/geekplux">{youtube?.subscriberCount || 'fetching...'}</a></li>
    <li>YouTube Views - <a href="https://youtube.com/c/geekplux">{youtube?.viewCount || 'fetching...'}</a></li>
  </ul>)
}
